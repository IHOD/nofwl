```bash
# branch rename
git branch -m <old_branch> <new_branch>

# rewriting history
git commit --amend

git fetch

# pull: fetch + merge
git pull

git tag <tagname>

# Cherry picking is the act of picking a commit from a branch and applying it to another.
git cherry-pick
```
